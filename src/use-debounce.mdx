# use-debounce

<Playground>
  {() => {
    const debouncedChange = useDebounce((v) => {
      console.log('debounced value', v);
    }, 500, { maxWait: 2000 });

    return (
      <input
        type="text"
        onChange={(e) => {
          debouncedChange(e.target.value)
        }}
      />
    )
  }}
</Playground>
